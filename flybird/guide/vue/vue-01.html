# MVVM
 - model:模型,数据对象(data)
 - view:视图,模板页面
 - viewModel: 视图模型(Vue的实例)

<!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8" />
		<title></title>
		<script src="js/vue.js" type="text/javascript" charset="utf-8"></script>
	</head>
	<body>
		<div id="test">
			<h2>双大括号表达式</h2>
			<p>{{ msg }}</p>    <!--textContent -->
			<p>{{ msg.toUpperCase() }}</p>
			<p v-html="site"></p>    <!--innerHTML -->
			<p v-text="msg">{{ msg }}</p> <!--textContent -->
			
			<h2>指令一: 强制数据绑定</h2>
			<!-- v-bind
			 或者:-->
			<img src='imurl'>
			<img v-bind:src="imgUrl">
			<img :src="imgUrl" /> <!--简洁语法 -->
			
			<h2>指令二: 绑定事件监听</h2>
			<button v-on:click="test">点击</button>
			<button @click="test2(msg.toUpperCase())">点击二</button>
		</div>
		<script>
			const vue = new Vue({
				el: '#test',
				data:{
					accName:"",
					msg:"i will go",
					site:"<a href='https://www.7miaoyu.com'>七秒鱼</a>"
				},
				methods:{
					test(){
						alert('hello')
					},
					test2(ad){
						alert(ad)
					}
				}
			})
		</script>
	</body>
</html>
