# 包装类

# 基本数据类型
 - String Number Boolean Null Undefined
# 引用数据类型
 - Object 

在Js中为我们提供了三个包装类,通过这三个包装类可以将基本数据类型的数据转换为对象
- String()
  - 可以将基本数据类型字符串转换为String对象

- Number()
  - 可以将基本数据类型的数字转换为Number对象 
- Boolean()
  - 可以将基本数据类型转换为Boolean对象
- 但是在实际使用过程中不会使用基本数据类型的对象
    如果使用基本数据类型的对象时候,在做一些比较时可能会带来一些不可预期的结果
- 方法和属性只能添加对象,不能添加给基本数据类型
 - 当对一些基本数据类型的值去调用属性和方法时,浏览器会临时使用包装类将其转换为对象,然后再调用对象的属性和方法,调用完以后,再将其转换为基本数据类型

# 字符串方法

# 在底层字符串是以字符数组的形式保存的
 - ["H","e"] 
# length属性
 - 可以获取字符串的长度
# charAt()
 - 可以返回字符串中指定位置的字符
 - 根据索引来获取指定字符
# charCodeAt()
 - 获取指定位置符的Unicode的编码
# fromCharCode
 - 可以根据字符编码去获取字符
 - 语法: String.fromCharCode()
# concat()
 - 可以用来连接两个或多个字符串
 - 作用和+一样
 - 对原字符串不会变
# indexof()
 - 该方法可以检索一个字符串中是否含有指定的内容
 - 如果该字符串中含有该内容,则会返回其第一次出现的索引
 - 如果没有找到指定的内容,则返回-1
 - 可以指定一个第二个参数,指定开始查找的位置
# lastIndexOf()
 - 该方法的作用和indexOf一样
 - 不同的是从后往前找
# slice
 - 可以从字符串中截取指定的内容
 - 不会影响原字符串,而是将截取到的内容返回
 - 参数:
   第一个,开始位置的索引(包括开始位置)
   第二个,结束位置的索引(不包括结束位置)
 - 第二个参数可以省略,直接截取后面的
 - 也可以传递一个负数 ,从后面开始计算
# subString()
 - 可以用来截取一个字符串,和slice()类似
 - 参数:
   - 第一个,开始截取位置的索引(包括开始位置)]
   - 第二个,结束位置的是(不包括结束位置)
 - 不同的是,这个方法不能接受负值作为参数
  传入负值,则默认0
 - 自动调整参数的位置,如果第二个参数小于第一个,则自动交换

# 正则表达式和正则相关方法

# split()
 - 可以将一个字符串拆分成一个数组
 - 根据任意字母来将字符串拆分
 - 方法中可以传递一个正则表达式作为参数,这样方法将会根据正则表达式去拆分字符串

# search() 相当于indexof
 - 搜索字符串中是否含有指定内容
 - 如果搜索到指定内容,则会返回第一次出现的索引,如果没有搜索到返回-1 
 - 它可以接受一个正则表达式作为参数,然后会根据正则表达式去检索字符串
# match()
 - 可以根据正则表达式,从一个字符串中将符合条件的内容提取出来
 - 默认情况下我们的match只会找到第一个符合要求的内容,找到以后就停止检索
 - 我们可以设置正则表达式作为全局匹配模式,这样就会匹配到所有的内容
 - 可以为正则表达式设置多个匹配模式,且顺序无所谓
 - match()会将匹配到的内容封装到一个数组中返回,即使只查到一个结果
# replace()
 - 可以将字符串中指定内容替换为新的内容
 - 参数: 
       1. 被替换的内容
       2. 新的内容
 - 默认只替换第一个

//split
var result = str.split(/[A-z]/);//根据正则表达式去拆分字母

//search
var str = "hello abc world";
result = str.search(/a[bef]c/);
console.log(result) //6

//match()
str = "1a2b3c4d5e6f7";
result = str.match(/[A-z]/);
console.log(result);//a

//replace()
str = "1a2b3c4d5e6f7";
result = str.replace("a","@_@");
console.log(result);//1@_@2b3c4d5e6f7

# 正则表达式语法

# 量词 
 - 通过量词可以设置一个内容出现的次数
 - 量词只对前边的内容起作用
 - {n} 正好出现n次
 - {m,n} m到n次
 - {m,} m次以上
 - + 表示至少一个 ,相当于 {1,}
 - * 0个或多个 相当于{0,}
 - ? 0个或一个 相当于{0,1}
 - ^ 表示开头
 - $ 表示结尾

//创建一个正则表达式检查一个字符串中是否含有aaa

var reg= /a{3}/;
var reg2= /(ab){3}/;
console.log(reg2.test("ababab"));//true
console.log(reg.test("aaabc"));//true

//+
var reg3 = /ab+c/;
console.log(reg3.test("abc"))

//*
var reg4 = /ab*c/;
console.log(reg4.test("abbbbbbc"));
// ?

var reg5 = /ab?c/;
console.log(reg5.test("abc"));

//检查一个字符串是否含有a开头
reg6 = /^a/;
console.log(reg6.test("abcabc"))//true
//匹配结尾的a
reg7 = /a$/;
console.log(reg6.test("abcaba"))//true
//如果在正则表达式中同时使用 ^ $ 则要求字符串必须完全符合正则表达式

//创建一个正则表达式是否是手机号
/*
  手机号规则:
        1. 1
        2. 3-9
        3. 任意输入

*/
var reg3 = /^1[3-9][0-9]{9}$/;
var phoneNum  ='17345952547';
console.log(reg3.test(phoneNum))
