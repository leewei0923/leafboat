# 自定义遍历数据

const banji = {
    name:"终极一班",
    stus:[
        '小明',
        '小红',
        '小菜',
        '小张'
    ],

[Symbol.iterator](){
    let index = 0;
    let _this = this;
    return {
        next: function(){
        if(index < _this.stus.length){
            
             const result = {value: _this.stus[index], done: false }
            index++;
            return result;
         }else {
             return {value :undefined, done:true}
         }
        }
    };
}
}

for(let v of banji){
    console.log(v)
}

# Promise 多个文件内容的获取

初步来看的话,感觉代码其实也不小,还是难以理解.我想这个可以避免回调地狱,避免无限的缩进.

const fs = require('fs');

const p = new Promise((resolve, reject) => {
    fs.readFile('../resources/名言.txt', (err, data) => {
        resolve(data);
        
    });
});

p.then(value => {
    return new Promise((resolve,reject)=>{
        fs.readFile('../resources/名言2.txt',(err,data)=>{
            resolve([value,data]);
        });
    })
}).then(value=>{
    return new Promise((resolve ,reject)=>{
        fs.readFile('../resources/名言3.txt',(err,data)=>{
            value.push(data);
            resolve(value);
        });
    });
}).then(value => {
    console.log(value.join('\r\n'))
})
